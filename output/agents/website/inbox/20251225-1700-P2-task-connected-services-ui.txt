## API Routes (Already Exist)

```typescript
// List connected services
GET /api/auth/connected-services
Response: { services: [{ provider: string, connected: boolean, username?: string }] }

// Initiate OAuth connection
POST /api/auth/connect/[provider]
Response: { redirectUrl: string }

// Disconnect service
DELETE /api/auth/disconnect/[provider]
Response: { success: boolean }
```

## Component Structure

```tsx
// ConnectedServicesPanel.tsx
export function ConnectedServicesPanel() {
  const { services, isLoading, error } = useConnectedServices();
  
  return (
    <div className="space-y-3">
      {services.map(service => (
        <ServiceCard
          key={service.provider}
          provider={service.provider}
          connected={service.connected}
          username={service.username}
          onConnect={() => handleConnect(service.provider)}
          onDisconnect={() => handleDisconnect(service.provider)}
        />
      ))}
    </div>
  );
}
```
